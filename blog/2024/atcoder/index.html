typedef long long ll;
const int maxn = 1e5;
int N,pos;
ll K,acc,m;
struct basket{
  ll val;
  int id;
}a[maxn + 5];

bool cmp2(basket x, basket y)
{
  return x.id < y.id;
}

bool cmp(basket x, basket y)
{
  if(x.val != y.val) return x.val < y.val;
  return x.id < y.id;
}

ll bisearch(ll nn,ll R,ll tot)
{
  ll L = 0;ll mid = 0;ll res = 0;
  for(;L <= R;)
  {
    mid = (L + R) >> 1;
    if(mid * nn <= tot)
    {
      res = mid;
      L = mid + 1;
    }
    else R = mid - 1;
  }
  return res;
}

int main()
{
  cin >> N >> K;

    cin >> a[i].val;

    a[i].val -= acc;
    lf = N - i + 1;
